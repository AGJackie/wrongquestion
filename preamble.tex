\cs_new_protected:Nn \ques_process_date:n {
  \noindent
  \begin{minipage}[t]{\dim_use:N \l_left_dim}

    \begin{tabular}{c}

      \endgroup
    \end{tabular}
  }
  \end{minipage}
}

\cs_new_protected:Nn \ques_label:n {
  \DTMsavedate{recent}{\l_recentdate_tl}
  \DTMsavenow{today}
  \DTMsaveddatediff{today}{recent}{\datediff}
  \clist_map_inline:Nn \l_review_point_clist
    {
      \int_compare:nT {\the\datediff == ##1}
      {
        \clist_gput_right:Nx \g_review_clist {\exp_not:N \ref{ques:#1}}
      }
    }
}

\NewDocumentEnvironment {ques} { +O{} +b }
{
  \clist_clear:N \l_date_clist
  \keys_set:nn {ques} {#1}
  \int_gincr:N \g_ques_count_int
  \clist_if_empty:NF \l_date_clist
    {
      \clist_get:NN \l_date_clist \l_recentdate_tl
    }
  \customlabel{ques\c_colon_str\int_use:N \g_ques_count_int}{\int_use:N \g_ques_count_int}
  \phantomsection
  \addcontentsline{toc}{subsection}{题目\int_use:N \g_ques_count_int}
  \ques_label:x {\int_use:N \g_ques_count_int}
  \par\noindent\ques_process_date:x {\int_use:N \g_ques_count_int}
  \begin{minipage}[t]{\dim_eval:n {\textwidth-\l_left_dim}}
    \noindent\textit{\Large\bfseries\int_use:N \g_ques_count_int}.\hspace{5pt}\ignorespaces#2%
  \end{minipage}
  \vskip\l_after_dim
} {}

}
